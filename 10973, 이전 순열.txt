#include <iostream>
#include <algorithm>

int arr[10001];
int n;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &arr[i]);
	}

	bool flag = false;
	flag = std::prev_permutation(arr, arr + n);
	
	if (flag)
		for (int i = 0; i < n; ++i) 
			printf("%d ", arr[i]);
	
	else printf("-1\n");
}
